#include "pico/stdlib.h" ---> In the sdk its pico/ thats why
#include "pico/cyw43_arch.h"

int main() {
    stdio_init_all();
    if (cyw43_arch_init()) {
        printf("Wi-Fi init failed");
        return -1;
    }
    while (true) {
        cyw43_arch_gpio_put(CYW43_WL_GPIO_LED_PIN, 1);
        sleep_ms(250);
        cyw43_arch_gpio_put(CYW43_WL_GPIO_LED_PIN, 0);
        sleep_ms(250);
    }
}

1️⃣ Headers

#include "pico/stdlib.h"
#include "pico/cyw43_arch.h"

  • pico/stdlib.h – standard library for GPIO, sleep, printf, etc.

  • pico/cyw43_arch.h – provides drivers for the CYW43 Wi-Fi chip on the Pico W.

    • Also includes functions to control the onboard LED (which is tied to the Wi-Fi chip)

2️⃣ stdio_init_all();

  • Initializes USB serial output and UART

  • Lets you use printf to see messages on your host computer

3️⃣ cyw43_arch_init()

if (cyw43_arch_init()) {
    printf("Wi-Fi init failed");
    return -1;
}

  • Initializes the CYW43 Wi-Fi chip

  • Returns non-zero if initialization fails

  • On the Pico W, the onboard LED is part of the CYW43 Wi-Fi chip hardware, so you have to init it even if you only want to blink the LED

4️⃣ Blink loop

while (true) {
    cyw43_arch_gpio_put(CYW43_WL_GPIO_LED_PIN, 1);
    sleep_ms(250);
    cyw43_arch_gpio_put(CYW43_WL_GPIO_LED_PIN, 0);
    sleep_ms(250);
}

  • CYW43_WL_GPIO_LED_PIN is a constant defined in cyw43_arch.h — it represents the onboard Wi-Fi LED

  • cyw43_arch_gpio_put(pin, value) sets the pin high (1) or low (0)

  • Sleep 250 ms between toggles → blinks LED at 2 Hz

5️⃣ Why Wi-Fi is involved

  • The Pico W’s onboard LED isn’t connected directly to a normal GPIO like the original Pico

  • It’s wired through the Wi-Fi chip (CYW43)

  • To control it, you must initialize the Wi-Fi driver first

  • So even a “simple blink” requires calling cyw43_arch_init() and using cyw43_arch_gpio_put() instead of gpio_put()

✅ TL;DR:

  • On Pico W, the LED is part of the Wi-Fi chip

  • Must init Wi-Fi chip even if you’re not connecting to Wi-Fi

  • cyw43_arch_gpio_put() controls that LED

  • Functionally, it blinks an LED just like normal, but uses the Wi-Fi driver API
